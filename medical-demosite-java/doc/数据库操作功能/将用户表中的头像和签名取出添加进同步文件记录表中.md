#将用户表中的头像和签名取出添加进同步文件记录表中

```sql
INSERT INTO md_sync_file (image_url, option_type, `status`, createdate)
SELECT avatar, 1, 0, NOW()
FROM sys_user
WHERE avatar IS NOT NULL
  AND avatar != ''
UNION
SELECT picture, 1, 0, NOW()
FROM sys_user
WHERE picture IS NOT NULL
  AND picture != ''
UNION
SELECT sign_pic, 1, 0, NOW()
FROM sys_user
WHERE sign_pic IS NOT NULL
  AND sign_pic != '';
```
